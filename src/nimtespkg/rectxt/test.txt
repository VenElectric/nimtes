    result = ACTI()
    consumeTag(s)
    size = readSize(s)
    skip(s,4)

proc readACTI(s,a:ACTI) =
    consumeTag(s)
    a.size = readSize(s)
    skip(s,4)
    a.flags = readUint32()
    while true:
        case peekTag(s):
            of NAME:
                readField(s,a.NAME)
            of MODL:
                readField(s,a.MODL)
            of FNAM:
                readField(s,a.FNAM)
            of SCRI:
                readField(s,a.SCRI)
            else: break
