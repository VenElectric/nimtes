BIPEDOBJECT requires check on size (used for multiple record types)
Maybe we can also do something similar for nested objects
proc readBipedObject(s;tag:string): BipedObject =
    checkTag(s.readTag(), tag)
    result = BipedObject()
    let size = s.readUint32()
    result.size = size
    if size == SZ8:
        result.kind = s.readUint8()
    if size == SZ32:
        result.kind = s.readUint32()
    var tag = s.peekTag()
    if tag == BNAM:
        result.bnam = s.readStrField(BNAM).some
    tag = s.peekTag()
    if tag == CNAM:
        result.cnam = s.readStrField(CNAM).some
